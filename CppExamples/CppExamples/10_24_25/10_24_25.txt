string getLocalTimeStr()
{
	time_t timer = time(nullptr);
	auto p = localtime(&timer);
	ostringstream oss;
	oss << setfill('0');

	oss << setw(2) << p->tm_mday << "_" << setw(2) << p->tm_mon + 1 << "_" << p->tm_year + 1900 << "_" <<
		setw(2) << p->tm_hour << '_' << setw(2) << p->tm_min << '_' << setw(2) << p->tm_sec;
	return oss.str();
	
}

			break;
		ifs.seekg((n - 1) * sizeof(int), ios::beg);
		ifs.read(reinterpret_cast<char *>(a), 10 * sizeof(int));
		for (auto x : a)
			cout << x << " ";
		cout << endl;
	}

	return 0;
}

int main(int argc, char *argv[])
{
	if (argc != 2) {
		cerr << "usage : <boat> <file name>" << endl;
		return 1;
	}
	char destname[L_tmpnam];
	tmpnam(destname);

	ifstream ifs(argv[1]);
	if (!ifs) {
		cout << argv[1] << " dosyasi acilamadi " << endl;
		return 2;
	}
	ofstream ofs(destname);
	if (!ofs) {
		cout << "gecici dosya olusturulamadi" << endl;
		return 3;
	}
	string sline;

	auto f = [](char c) {return !isspace(c);};

	
	while (getline(ifs, sline)) {
		auto iter = find_if(sline.begin(), sline.end(), f);
		auto riter = find_if(sline.rbegin(), sline.rend(), f);
		sline.assign(iter, riter.base());
		sline.erase(unique(sline.begin(), sline.end(),
			[](char c1, char c2) {return c1 == ' ' && c2 == ' ';}), sline.end());
		ofs << sline << endl;
	}

	ifs.close();
	ofs.close();

	if (remove(argv[1])) {
		cerr << "dosya silinemiyor" << endl;
		return 4;
	}

	if (rename(destname, argv[1])) {
		cerr << "dosya ismi degistirilemiyor" << endl;
		return 5;
	}

	return 0;
}
