void displayProperties(const unordered_set<string> &us)
{
	cout << "size              : " << us.size() << endl;
	cout << "bucket size       : " << us.bucket_count() << endl;
	cout << "load factor       : " << us.load_factor() << endl;
	cout << "max load factor   : " << us.max_load_factor() << endl;



	for (size_t i = 0; i < us.bucket_count(); ++i) {
		cout << "bucket[" << i << "] (" << us.bucket_size(i) << ") = ";
		for (auto iter = us.cbegin(i); iter != us.cend(i); ++iter)
			cout << *iter << " ";
		cout << endl;
	}

	cout << sl;
}

	displayProperties(names);

	cout << names.bucket("suleyman") << endl;

	return 0;
